CREATE TABLE public.estaciondeservicio (
    nombre character varying(45) NOT NULL,
    factorutilidad double precision DEFAULT 0.01,
    precio93 integer NOT NULL,
    precio95 integer NOT NULL,
    precio97 integer NOT NULL,
    preciodiesel integer NOT NULL,
    preciokerosene integer NOT NULL
);

CREATE TABLE public.surtidor (
    nombre character varying(45) NOT NULL,
    tipo character varying(45) NOT NULL,
    precio integer NOT NULL,
    litrosconsumidos double precision NOT NULL,
    litrosdisponibles double precision NOT NULL,
    cargasrealizadas integer NOT NULL,
    refestacion character varying(45)
);

CREATE TABLE public.ventas (
    refsurtidor character varying(45),
    precio integer NOT NULL,
    litros double precision NOT NULL
);
